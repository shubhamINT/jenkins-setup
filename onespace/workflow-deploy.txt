pipeline {
    agent any

    environment {
        SERVER_HOST  = 'developer@10.10.3.83'
        PROJECT_DIR  = '~/workflow_backend'
        SSH_CREDS_ID = 'ssh-key'
    }

    stages {

        stage('Deploy Workflow Backend') {
            steps {
                sshagent([env.SSH_CREDS_ID]) {
                    sh """
                        echo "▶ Deploying Workflow Backend on ${env.SERVER_HOST}..."
                        ssh -o StrictHostKeyChecking=no ${env.SERVER_HOST} "
                            cd ${env.PROJECT_DIR} &&
                            bash deploy.sh
                        "
                    """
                }
            }
        }

    }

    post {
        success {
            echo "✅ Workflow Backend deployed successfully."
        }
        failure {
            echo "❌ Deployment failed — check Console Output."
        }
    }
}
