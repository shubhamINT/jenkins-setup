pipeline {
    agent any

    environment {
        SERVER_HOST  = 'developer@10.10.3.83'
        PROJECT_DIR  = '~/onespace_docs_agent_backend'
        SSH_CREDS_ID = 'ssh-key'
    }

    stages {

        stage('Deploy Onespace RAG Pipeline') {
            steps {
                sshagent([env.SSH_CREDS_ID]) {
                    sh """
                        echo "▶ Deploying Onespace RAG Pipeline on ${env.SERVER_HOST}..."
                        ssh -o StrictHostKeyChecking=no ${env.SERVER_HOST} "
                            echo '🧹 Pruning unused Docker resources...' &&
                            docker system prune -a -f &&
                            cd ${env.PROJECT_DIR} &&
                            echo 'Pulling the latest code...' &&
                            git pull origin main &&
                            echo 'Deploying the service...' &&
                            bash deploy-rag-pipeline.sh
                        "
                    """
                }
            }
        }

    }

    post {
        success {
            echo "✅ Onespace RAG Pipeline deployed successfully."
        }
        failure {
            echo "❌ Deployment failed — check Console Output."
        }
    }
}
