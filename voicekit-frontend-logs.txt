pipeline {
    agent any

    environment {
        SERVER_HOST  = 'developer@13.234.150.174'  // has: intvyom-api
        SSH_CREDS_ID = 'ssh-key'
        CONTAINER    = 'intvyom-frontend'
    }

    parameters {
        choice(
            name: 'DURATION',
            choices: ['1', '2', '5'],
            description: 'How many minutes to stream logs'
        )
    }

    stages {
        stage('Live Logs') {
            steps {
                script {
                    sshagent([env.SSH_CREDS_ID]) {
                        sh """
                            echo "▶ Streaming ${env.CONTAINER} on ${env.SERVER_HOST} for ${params.DURATION} min(s)..."
                            ssh -o StrictHostKeyChecking=no ${env.SERVER_HOST} \
                                "timeout ${params.DURATION}m docker logs -f --tail=50 ${env.CONTAINER} 2>&1 | sed 's/^/[intvyom-api] /'" || true
                        """
                    }
                }
            }
        }
    }

    post {
        success {
            echo "✅ Logs streamed successfully for ${params.DURATION} minute(s)"
        }
        failure {
            echo "❌ Log streaming failed — check SSH / Docker / container name"
        }
        aborted {
            echo "⚠️ Stopped manually by user"
        }
    }
}
