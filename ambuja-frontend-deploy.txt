pipeline {
    agent any

    environment {
        SERVER_HOST  = 'developer@13.205.194.139'
        PROJECT_DIR  = '~/Ambuja_ai_recruiter/frontend'
        PM2_APP_NAME = 'ambuja_frontend'
        SSH_CREDS_ID = 'ssh-key'
        PORT = '3000'
    }

    stages {

        stage('Deploy Ambuja Frontend') {
            steps {
                sshagent([env.SSH_CREDS_ID]) {
                    sh """
                        echo "▶ Deploying ${env.PM2_APP_NAME} on ${env.SERVER_HOST}..."
                        ssh -o StrictHostKeyChecking=no ${env.SERVER_HOST} "
                            source ~/.nvm/nvm.sh &&
                            cd ${env.PROJECT_DIR} &&
                            pm2 restart ${env.PM2_APP_NAME} 2>/dev/null || pm2 start \"npx serve -s dist --single -l ${env.PORT}\" --name ${env.PM2_APP_NAME} &&
                            pm2 status ${env.PM2_APP_NAME}
                        "
                    """
                }
            }
        }

    }

    post {
        success {
            echo "✅ ${env.PM2_APP_NAME} deployed successfully."
        }
        failure {
            echo "❌ Deployment failed — check Console Output."
        }
    }
}
